DML(Data Manipulation Language)

-SELECT : 조회(검색)
	SELECT 컬럼명1, 컬럼명2,.....FROM 테이블명
	WHERE 조건식;

-INSERT : 추가
	INSERT INTO 테이블명
	(컬럼명1, 컬럼명2,...)
	VALUES(값1, 값2,...);

	INSERT INTO 테이블명 VALUES(값1, 값2,...)
-UPDATE : 수정
	UPDATE 테이블명
	SET 기존컬럼명 = 새로운 값
	WHERE 조건식

-DELETE : 삭제
	DELETE FROM 테이블명
	WHERE 조건식;
